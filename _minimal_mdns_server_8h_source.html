26 template <typename T>
27 bool IsCurrentInterfaceUsable(T & iterator)
28 {
29  if (!iterator.IsUp() || !iterator.SupportsMulticast())
30  {
31  return false; // not a usable interface
32  }
33  char name[chip::Inet::InterfaceIterator::kMaxIfNameLength];
34  if (iterator.GetInterfaceName(name, sizeof(name)) != CHIP_NO_ERROR)
35  {
36  ChipLogError(Discovery, "Failed to get interface name.");
37  return false;
38  }
39 
40  // TODO: need a better way to ignore local loopback interfaces/addresses
41  // We do not want to listen on local loopback even though they are up and
42  // support multicast
43  //
44  // Some way to detect 'is local looback' that is smarter (e.g. at least
45  // strict string compare on linux instead of substring) would be better.
46  //
47  // This would reject likely valid interfaces like 'lollipop' or 'lostinspace'
48  if (strncmp(name, "lo", 2) == 0)
49  {
51  return false;
52  }
53  return true;
54 }
